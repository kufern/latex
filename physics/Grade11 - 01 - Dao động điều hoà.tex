\section{Những đặc điểm của dao động cơ}
	\begin{itemize}
		\item Có một vị trí cân bằng xác định
		\item Vật chuyển động qua lại quanh vị trí cân bằng đó
		\item Dao động cơ của một vật có thể là \textit{tuần hoàn} hay \textit{không tuần hoàn}:
		\begin{itemize}[label=\textopenbullet]
			\item Dao động tuần hoàn: Sau những khoảng thời gian bằng nhau, vật trở lại vị trí cũ theo hướng cũ
			\item Dao động không tuần hoàn
	\end{itemize}
\end{itemize}
