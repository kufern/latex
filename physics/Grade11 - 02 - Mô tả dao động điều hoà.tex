\section{Pha dao động}
	\vspace{-13pt}
	\begin{center}
		\[ \boxed{\phi = \omega t + \varphi} \]
		\(\Rightarrow \phi\) là hàm số bậc nhất theo \(t\)
		\begin{empheq}[box={\custombox[2pt][2pt]}]{equation*}
			\: \omega = \tan \left( \frac{\phi}{t} \right)
		\end{empheq}
	\end{center}
	\vspace{3pt}
	\begin{center}
		\textcolor{red}{\large \underline{\textbf{Chú ý:}} \( -\pi \leq \phi \leq \pi\)}
	\end{center}

\section{Độ lệch pha}
	\subsection{Độ lệch pha của một dao động tại hai thời điểm}
	\vspace{5pt}
	\[\begin{aligned}
		\Delta \phi &= \phi_2 - \phi_1 \\
		&= (\omega t_2 + \varphi) - (\omega t_1 + \varphi) \\
		&= \omega t_2 - \omega t_1 \\
		&= \omega (t_2 - t_1) \\
		&= \omega \times \Delta t
	\end{aligned}\]
	\[ \Rightarrow \boxed{\Delta \phi = \omega \times \Delta t} \qquad \]

	\subsection{Độ lệch pha của hai dao động cùng phương, cùng tần số ở một thời điểm}
	\vspace{5pt}
	\[\begin{aligned}
		\Delta \phi &= \phi_2 - \phi_1 \\
		&= (\omega t + \varphi_2) - (\omega t + \varphi_1) \\
		&= \varphi_2 - \varphi_1
	\end{aligned}\]
	\[ \Rightarrow \boxed{\Delta \phi = \varphi_2 - \varphi_1} \qquad \]
	
	\begin{center}
		\vspace{3pt}
		\textcolor{red}{\underline{\textbf{Chú ý:}}\hspace{5pt}
			\(\begin{aligned}[t]
				\varphi_1 &> \varphi_2 \Rightarrow \text{Dao động 1 sớm pha hơn dao động 2} \\
				\varphi_1 &< \varphi_2 \Rightarrow \text{Dao động 1 trễ pha hơn dao động 2} \\
				\varphi_1 &= \varphi_2 \Rightarrow \text{Hai dao động cùng pha (đồng pha)} \\
				\varphi_1 &= \varphi_2 \pm \pi \Rightarrow \text{Dao động 1 ngược pha với dao động 2} \\
			\end{aligned}\)}
	\end{center}

	\vspace{-1.5cm}

			\: \varphi = \pm \arccos \left( \frac{x_0}{A} \right)
		\end{empheq}
	\end{center}
