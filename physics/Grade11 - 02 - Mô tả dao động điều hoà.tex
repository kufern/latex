\section{Các đại lượng đặc trưng của dao động \\ điều hoà}
	\subsection{Biên độ}
		\vspace{5pt}
		\begin{itemize}
			\item Độ dịch chuyển cực đại của vật tính từ vị trí cân bằng
			\item Ký hiệu: \(A\)
			\item Luôn dương
			\item Không phụ thuộc vào \(t\)
			\item Phụ thuộc vào điều kiện kích thích
			\item Đơn vị: m, cm, mm,\,\ldots
		\end{itemize}

	\subsection{Tần số góc}
		\vspace{5pt}
		\begin{itemize}
			\item Ký hiệu: \(\omega\)
			\item Luôn dương
			\item Không phụ thuộc vào \(t\)
			\item Phụ thuộc vào đặc tính của hệ, môi trường,\,\ldots
			\item Đơn vị: rad/s
		\end{itemize}

	\subsection{Chu kỳ}
		\vspace{5pt}
		\begin{itemize}
			\item Khoảng thời gian để vật thực hiện một dao động toàn phần
			\item Ký hiệu: \(T\)
			\item Phụ thuộc vào đặc tính của hệ, môi trường,\,\ldots
			\item Đơn vị: s
			\pagebreak
			\item Công thức:
		\end{itemize}
			\begin{minipage}[t]{0.5\textwidth}
				\vspace{-12pt}
				\[ \qquad\quad \boxed{T = \frac{\Delta t}{N} = \frac{2 \pi}{\omega}} \]
			\end{minipage}
			\begin{minipage}[t]{0.49\textwidth}
				\vspace{-10pt}
				\(\Delta t\): Thời gian mà dao động xảy ra (s)
				\\ \(N\): Số dao động mà xảy ra trong thời gian đó
			\end{minipage}
			\vspace{4pt}

	\subsection{Tần số}
		\vspace{5pt}
		\begin{itemize}
			\item Số dao động mà vật thực hiện trong một giây
			\item Ký hiệu: \(f\)
			\item Phụ thuộc vào đặc tính của hệ, môi trường,\,\ldots
			\item Đơn vị: Hz \( (= \dfrac{1}{s} = s^{-1}) \)
			\item Công thức:
				\vspace{6pt}
				\[ \boxed{f = \frac{1}{T} = \frac{N}{\Delta t} = \frac{\omega}{2 \pi}} \]
		\end{itemize}

\section{Pha dao động}
	\vspace{-13pt}
	\begin{center}
		\[ \boxed{\phi = \omega t + \varphi} \]
		\(\Rightarrow \phi\) là hàm số bậc nhất theo \(t\)
		\[ \boxed{ \: \omega = \tan \left( \frac{\phi}{t} \right)} \! \]
	\end{center}
